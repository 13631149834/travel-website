使用方法

1. 引入文件

<link rel="stylesheet" href="css/share.css">
<script src="js/share.js"></script>

2. 在页面中添加分享容器

<!-- 自动初始化，data-type 指定分享类型 -->
<div class="share-container" data-type="guide" data-id="123"></div>

<!-- 或手动初始化 -->
<div class="share-buttons" id="my-share"></div>

3. JavaScript 初始化

// 自动初始化（容器需有 share-container 类）
// 页面加载后自动生成按钮并绑定事件

// 手动初始化
ShareModule.init('#my-share', {
  url: 'https://example.com/page',
  title: '页面标题',
  description: '页面描述',
  image: 'images/cover.jpg'
});

// 或直接调用分享方法
ShareModule.shareToWechat({ url: location.href, title: '标题' });
ShareModule.shareToWeibo({ url: location.href, title: '标题' });
ShareModule.shareToQQ({ url: location.href, title: '标题' });
ShareModule.copyLink({ url: location.href });

// 为详情页生成分享数据
const data = ShareModule.generateShareData('guide', { id: '123', name: '导游名称' });

// 获取分享统计
const stats = ShareModule.getStats();
console.log(stats); // { wechat: 0, weibo: 0, qq: 0, copy: 0, total: 0 }
